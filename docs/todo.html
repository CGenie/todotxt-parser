class Todo
    constructor: (params) ->
        items_ = (TodoItem.parse text for text in (params?.items || []))
        @items = (item for item in items_ when item.validate())

    render: () ->
        (item.render() for item in this.items).join "\n"

    this.parse = (text) ->
        text ?= ''

        new Todo
            items: text.split('\n')
